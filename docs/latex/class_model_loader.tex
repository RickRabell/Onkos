\doxysubsection{Detailed Description}
A utility class for loading 3D model data from files. 

\begin{DoxyAuthor}{Author}
Ricardo Rabell 
\end{DoxyAuthor}
\begin{DoxyDate}{Date}
2025-\/11-\/05
\end{DoxyDate}
